76 void Fp2_mul(Fp2 *ANS,Fp2 *A,Fp2 *B){
77  Fp tmp1,tmp2,tmp3,tmp4;
78  Fp_init(&tmp1);
79  Fp_init(&tmp2);
80  Fp_init(&tmp3);
81  Fp_init(&tmp4);
82 
83  //set
84  Fp_mul(&tmp1,&A->x0,&B->x0);//a*c
85  Fp_mul(&tmp2,&A->x1,&B->x1);//b*d
86  Fp_add(&tmp3,&A->x0,&A->x1);//a+b
87  Fp_add(&tmp4,&B->x0,&B->x1);//c+d
88  //x0
89  Fp_mul_basis(&ANS->x0,&tmp2);//b*d*v
90  Fp_add(&ANS->x0,&ANS->x0,&tmp1);//a*c+b*d*v
91  //x1
92  Fp_mul(&ANS->x1,&tmp3,&tmp4);//(a+b)(c+d)
93  Fp_sub(&ANS->x1,&ANS->x1,&tmp1);
94  Fp_sub(&ANS->x1,&ANS->x1,&tmp2);
95 
96  //clear
97  Fp_clear(&tmp1);
98  Fp_clear(&tmp2);
99  Fp_clear(&tmp3);
100  Fp_clear(&tmp4);

342 void Fp2_sqrt(Fp2 *ANS,Fp2 *A){
343  Fp2 x,y,t,k,n,tmp;
344  Fp2_init(&x);
345  Fp2_init(&y);
346  Fp2_init(&t);
347  Fp2_init(&k);
348  Fp2_init(&n);
349  Fp2_init(&tmp);
350  unsigned long int e,m;
351  mpz_t exp,q,z,result;
352  mpz_init(exp);
353  mpz_init(q);
354  mpz_init(z);
355  mpz_init(result);
356  gmp_randstate_t state;
357  gmp_randinit_default(state);
358  gmp_randseed_ui(state,(unsigned long)time(NULL));
359 
360  Fp2_set_random(&n,state);
361  while(Fp2_legendre(&n)!=-1){
362  Fp2_set_random(&n,state);
363  }
364  mpz_pow_ui(q,curve_parameters.prime,2);
365  mpz_sub_ui(q,q,1);
366  mpz_mod_ui(result,q,2);
367  e=0;
368  while(mpz_cmp_ui(result,0)==0){
369  mpz_tdiv_q_ui(q,q,2);
370  mpz_mod_ui(result,q,2);
371  e++;
372  }
373  Fp2_pow(&y,&n,q);
374  mpz_set_ui(z,e);
375  mpz_sub_ui(exp,q,1);
376  mpz_tdiv_q_ui(exp,exp,2);
377  Fp2_pow(&x,A,exp);
378  Fp2_mul(&tmp,&x,&x);
379  Fp2_mul(&k,&tmp,A);
380  Fp2_mul(&x,&x,A);
381  while(Fp2_cmp_one(&k)!=0){
382  m=1;
383  mpz_ui_pow_ui(exp,2,m);
384  Fp2_pow(&tmp,&k,exp);
385  while(Fp2_cmp_one(&tmp)!=0){
386  m++;
387  mpz_ui_pow_ui(exp,2,m);
388  Fp2_pow(&tmp,&k,exp);
389  }
390  mpz_sub_ui(exp,z,m);
391  mpz_sub_ui(exp,exp,1);
392  mpz_ui_pow_ui(result,2,mpz_get_ui(exp));
393  Fp2_pow(&t,&y,result);
394  Fp2_mul(&y,&t,&t);
395  mpz_set_ui(z,m);
396  Fp2_mul(&x,&x,&t);
397  Fp2_mul(&k,&k,&y);
398  }
399  Fp2_set(ANS,&x);
400 
401  mpz_clear(exp);
402  mpz_clear(q);
403  mpz_clear(z);
404  mpz_clear(result);
405  Fp2_clear(&x);
406  Fp2_clear(&y);
407  Fp2_clear(&t);
408  Fp2_clear(&k);
409  Fp2_clear(&n);
410  Fp2_clear(&tmp);
411 }
