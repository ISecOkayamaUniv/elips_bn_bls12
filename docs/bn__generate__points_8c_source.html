52  EFp12 random_P,P,frobenius_P;
53  EFp12_init(&random_P);

59  EFp12_rational_point_bn(&random_P);
60  mpz_pow_ui(exp,curve_parameters.order,2);
61  mpz_tdiv_q(exp,curve_parameters.EFpd_total,exp);
62  EFp12_SCM(&P,&random_P,exp);
63  Fp12_frobenius_map_p1(&frobenius_P.x,&P.x);
64  Fp12_frobenius_map_p1(&frobenius_P.y,&P.y);
65  EFp12_set_neg(&P,&P);
66  EFp12_ECA(Q,&P,&frobenius_P);
67 
68  mpz_clear(exp);
69  EFp12_clear(&random_P);
70  EFp12_clear(&P);
71  EFp12_clear(&frobenius_P);
72 }
